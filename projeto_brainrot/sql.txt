CREATE TABLE conta (
	id INT PRIMARY KEY AUTO_INCREMENT,
    nome VARCHAR(50) NOT NULL
);

CREATE TABLE brainrot (
	id INT PRIMARY KEY AUTO_INCREMENT,
    id_conta INT,
    nome VARCHAR(50) NOT NULL,
    raridade VARCHAR(50) NOT NULL,
    efeito VARCHAR(200) NOT NULL,
    FOREIGN KEY (id_conta) REFERENCES conta(id)
);
